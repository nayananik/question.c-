#include <stdio.h>


int main()

{
   
   int n,a[100],c=1;
 
   int b[100];
 
   scanf("%d",&n);

    for(int i=0;i<n;i++)
 
   {
       
	 scanf("%d",&a[i]);
 
       
    }
   
 for(int i=0;i<n;i++)
 
   {
      
	 if(a[i]==c)
  
      {
           
	    b[i]=a[i];
  
          c++;
    
    }
      
	  else if(a[i]>c)
   
     {
    
        b[i]=a[i+1];
  
          c++;
      
  }
       
	 else
    
    {
        
	    b[i]=c;
 
           c++;
     
   }
   
 }
  
  for(int i=1;i<=n;i++)
 
   {
       
 if(b[i-1]==i)
    
        printf("happy");
  
      else
          
	  printf("sad");
  
  }
    

  
  return 0;

}
